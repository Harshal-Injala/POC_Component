Form Components

Prerequisite:

HTML

                <div class="container mt-5">
                    <h2 class="form-header">Form Header Goes Here</h2>
                    <form class="form-main">
                        <table>
                            <!-- Component Code Goes Here. -->
                        </table>
                    </form>
                </div>
CSS

            .form-header{
                text-transform: uppercase;
                text-align: center;
                font-weight: bold;
            }
            .form-main{
                margin-top: 15px;
                margin-bottom: 15px;
            }
            .form-main table{
                width: 60%;
                margin: 0px auto;
            }
            .form-main table tr td{
                padding: 10px;
            }
            .form-main table tr td input,.form-main table tr td select{
                width: 100%;
                height: 50px;
            }
